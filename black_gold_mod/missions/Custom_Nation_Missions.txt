industrial_mission = {
	slot = 1
	generic = yes
	ai = no
	#potential = {
	#	is_dynamic_tag = yes 
	#}
	
	promote_urbanization = {
		icon = industrialized_rhineland
		position = 6
		required_missions = {  }
		trigger = {
			has_institution = industrialization
			
			OR = {
				num_of_owned_provinces_with = {
					value = 6
					trade_goods = iron
				}
				num_of_owned_provinces_with = {
					value = 6
					trade_goods = cloth
				}
				num_of_owned_provinces_with = {
					value = 2
					trade_goods = industrial_equipment
				}
			}
			
			OR = {
				num_of_owned_provinces_with = {
					value = 3
					trade_goods = oil
				}
				num_of_owned_provinces_with = {
					value = 3
					trade_goods = coal
				}
				num_of_owned_provinces_with = {
					value = 3
					trade_goods = industrial_equipment
				}
			}
		}
		effect = {
			capital_scope = {
				add_base_tax = 3
				add_base_production = 3
				add_base_manpower = 3
				add_province_modifier = {
					name = "commitment_urbanization"
					duration = -1
				}
			}
		}
	}
	
	create_industrial_area = {
		icon = promote_urbanization
		required_missions = { promote_urbanization }
		position = 7
		trigger = { 
			OR = {
				any_owned_province = {
					trade_goods = coal
					base_production = 10
				}
				any_owned_province = {
					trade_goods = oil
					base_production = 10
				}
			}
		}
		effect = {
			random_owned_area = { 
				area = {
					limit = { owned_by = ROOT }
					add_province_modifier = {
						name = "industrial_area"
						duration = -1
					}
					add_base_tax = 2
					add_base_production = 2
					add_base_manpower = 2
				}
			} 
		}
	}
	
	industrial_economy = {
		icon = mission_rb_build_furnaces
		required_missions = { create_industrial_area }
		position = 8
		trigger = { 
			capital_scope = { 
				development = 50
			} 
			num_of_owned_provinces_with = {
				value = 4
				development = 30
			}
			current_age = age_of_industrialization
		}
		effect = {
			add_prestige = 25
			add_splendor = 200
			
			add_country_modifier = {
				name = "industrialized_economy"
				duration = -1
			}
		}
	}
	
	industrial_universities = {
		icon = Compulsory_Education
		required_missions = { industrial_economy }
		position = 9
		trigger = { 
			capital_scope = { 
				has_building = university
			} 
			
			num_of_owned_provinces_with = {
				value = 3
				development = 25
			}
		}
		effect = {
			capital_scope = {
				add_base_tax = 3
				add_base_production = 3
				add_base_manpower = 3
			}
			add_country_modifier = {
				name = "national_academia"
				duration = -1
			}
			
			add_prestige = 25
			add_splendor = 200
		}
	}
}
