soviet_revolution = {
	potential = {
		OR = {
			tag = RUS
			tag = RUF
		}
		is_subject = no
		NOT = {
			OR = {
				has_reform = futurist_mechanic
				has_reform = socialist_mechanic
				has_reform = fascist_mechanic
			}
			#has_country_modifier = victorious_against_revolution
		}
		adm_tech = 35
		has_global_flag = communist_manifesto_written
	}
	
	
	can_start = {
		OR = {
			current_age = age_of_great_wars
			current_age = age_of_industrialization
			war_exhaustion = 12
			NOT = { stability = 0 }
			NOT = {
				OR = {
					manpower_percentage = 0.675
					prestige = -15
				}
			}
		}
		is_institution_enabled = nationalism_institution
		has_any_disaster = no
	}
	
	can_stop = { 
		OR = {
			has_any_disaster = yes
		}
	}
	
	
	progress = {
		modifier = {
			factor = 1
			NOT = { prestige = 0 }
		}
		modifier = {
			factor = 2
			war_exhaustion = 5
		} 
		modifier = {
			factor = 2
			num_of_loans = 1
			NOT = { num_of_loans = 4 }
		}	
		modifier = {
			factor = 1
			absolutism = 70
		}
		modifier = {
			factor = 2
			absolutism = 80
		}
		modifier = {
			factor = 3
			num_of_loans = 4
		}
		modifier = {
			factor = 4
			is_bankrupt = yes
		}	
		modifier = {
			factor = -3
			religion_group = muslim
		}
	}
	
	can_end = {
		stability = 3
		NOT = { num_of_rebel_controlled_provinces = 1 }
	}
	

	modifier = {
		global_unrest = 40
		stability_cost_modifier = 0.5
		defensiveness = -0.5
	}	
	
	
	on_start = socialist_revolution.1
	on_end = socialist_revolution.20
	
	on_monthly = {
		events = {
			socialist_revolution.10
			socialist_revolution.15
		}
		random_events = { 
			1000 = 0
			100 = socialist_revolution.3
		}
	}
}
