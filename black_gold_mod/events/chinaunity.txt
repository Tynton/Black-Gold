namespace = chinaunity

#initial chinese revolt

country_event = {
	id = chinaunity.1
	title = "chinaunity.n.1"
	desc = "chinaunity.d.1"
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = yes
	
	major = yes

	trigger = {
		is_year = 1895
		AND = {
			NOT = {
				culture_group = east_asian
			}
			NOT = {
				exists = ROC
			}
			any_owned_province = {
				culture_group = east_asian
			}
			NOT = {
				tag = ROC
			}
			continent = asia
		}
	}	

	mean_time_to_happen = {
		months = 60
	}
	
	option = {
		name = "chinaunity.1.a"
		release = ROC
		ROC = {
			change_government = republic
			add_government_reform = oligarchy_reform
			add_government_reform = parliamentary_reform
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_independence_war
			}
		}
	}
	
	option = {
		name = "chinaunity.1.b"
		release = ROC
		ROC = {
			change_government = republic
			add_government_reform = oligarchy_reform
			add_government_reform = parliamentary_reform
			create_alliance = ROOT
		}
	}
}

#province revolt

province_event = {
	id = chinaunity.3
	title = "chinaunity.n.3"
	desc = "chinaunity.d.3"
	picture = ANGRY_MOB_eventPicture

	trigger = {
		has_institution = nationalism_institution
		is_year = 1895
		NOT = {
			is_year = 1930
		}
		culture_group = east_asian
		is_core = ROC
		NOT = {
			devastation = 1
		}
		owner = {
			NOT = {
				culture_group = east_asian
			}
		}
		exists = ROC
		continent = asia
	}	

	mean_time_to_happen = {
		months = 18
	}
	
	option = {
		name = "chinaunity.3.a"
		spawn_rebels = {
			type = nationalist_rebels
			size = 3
			friend = ROC
		}
		
		add_devastation = 15
	}
	option = {
		name = "chinaunity.3.b"
		cede_province = ROC
		add_core = ROC
	}
}

#nations join china

country_event = {
	id = chinaunity.4
	title = "chinaunity.n.4"
	desc = "chinaunity.d.4"
	picture = ANGRY_MOB_eventPicture

	trigger = {
		exists = ROC
		ai = yes 
		OR = {
			culture_group = east_asian
			is_subject_of = ROC
		}
		ROC = {
			is_subject = no
		}
		any_neighbor_country = {
			tag = ROC
		}
	}	

	mean_time_to_happen = {
		months = 9
	}
	
	option = {
		name = "chinaunity.4.a"
		ROC = {
			inherit = ROOT
			ROOT = {
				every_owned_province = {
					add_core = ROC
				}
			}
		}
	}
}
