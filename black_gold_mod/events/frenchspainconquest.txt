province_event = {
	id = frenchspainconquest.1
	title = "frenchspainconquest.n.1"
	desc = "frenchspainconquest.d.1"
	picture = ANGRY_MOB_eventPicture

	trigger = {
		AND = {
			culture = castillian
			owner = {
				culture_group = french
				#has_country_modifier = funding_cultural_assimilation
			}
		}
		owner = {
			OR = {
				stability = 2
				prestige = 60
				full_idea_group = expansion_ideas
			}
			government_rank = 3
		}
	}	

	mean_time_to_happen = {
		months = 240
	}
	
	option = {
		name = "frenchspainconquest.1.a"
		change_culture = castillan
	}
}
