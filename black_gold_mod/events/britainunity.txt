#province revolt

province_event = {
	id = britainunity.3
	title = "britainunity.n.3"
	desc = "britainunity.d.3"
	picture = ANGRY_MOB_eventPicture

	trigger = {
		is_year = 1815
		NOT = {
			current_age = age_of_great_wars
		}
		NOT = {
			devastation = 1
		}
		culture_group = british
		is_core = GBR
		owner = {
			NOT = {
				culture_group = british
			}
			NOT = {
				prestige = 65
			}
		}
		exists = GBR
		unrest = -2
	}	

	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = "britainunity.3.a"
		spawn_rebels = {
			type = nationalist_rebels
			size = 5
			friend = GBR
		}
		add_devastation = 15
	}
	option = {
		name = "britainunity.3.b"
		cede_province = GBR
		add_core = GBR
	}
}
