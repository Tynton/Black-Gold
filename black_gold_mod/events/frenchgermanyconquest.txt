province_event = {
	id = frenchgermanyconquest.1
	title = "frenchgermanyconquest.n.1"
	desc = "frenchgermanyconquest.d.1"
	picture = ANGRY_MOB_eventPicture

	trigger = {
		AND = {
			culture = swabian
			owner = {
				culture_group = french
				#has_country_modifier = funding_cultural_assimilation
			}
		}
		owner = {

	}

		change_culture = souabe
	}
}
