namespace = waldensian_religion

# Conversion to Waldensian
country_event = {
	id = waldensian.1
	title = waldensian.1.t
	desc = waldensian.1.d
	picture = PRAYING_eventPicture
	
	trigger = {
		religion = catholic
		OR = {
			num_of_owned_provinces_with = {
				value = 4
				religion = waldensian
			}
			
			dominant_religion = waldensian
		}
		
		NOT = {
			has_country_flag = no_waldensian
		}
	}
	
	mean_time_to_happen = {
		months = 48
	}
	
	option = { # we will convert
		name = "waldensian.1.a"
		add_stability = -2
		add_prestige = -50
		change_religion = waldensian
	}
	option = { # we will not convert to a perverted faith
		name = "waldensian.1.b"
		add_stability = -1
		add_prestige = 15
		set_country_flag = no_waldensian
	}
}

# Quest for Lyons
country_event = {
	id = waldensian.2
	title = waldensian.2.t
	desc = waldensian.2.d
	picture = PRAYING_eventPicture
	
	trigger = {
		religion = waldensian
		
		203 = {
			NOT = {
				religion = waldensian
				owned_by = ROOT
			}
		}
		
		NOT = {
			has_country_flag = no_lyons
		}
		
		any_neighbor_province = {
			province_id = 203
		}
	}
	
	mean_time_to_happen = {
		months = 24
	}
	
	option = { # we will claim Waldo's homeland
		name = "waldensian.2.a"
		add_stability = 1
		add_permanent_claim = 203
	}
	option = { # we will not
		name = "waldensian.2.b"
		add_stability = -1
		set_country_flag = no_lyons
	}
}
