namespace = frenchitalyconquest

province_event = {
	id = frenchitalyconquest.1
	title = "frenchitalyconquest.n.1"
	desc = "frenchitalyconquest.d.1"
	picture = ANGRY_MOB_eventPicture
	
	hidden = yes

	trigger = {
		AND = {
			culture = tuscan
			owner = {
				culture_group = french
				#has_country_modifier = funding_cultural_assimilation
			}
		}
		NOT = {
			local_autonomy = 5
			unrest = 1
		}
		owner = {
			OR = {
				stability = 2
				prestige = 50
			}
		}
		has_owner_accepted_culture = no
	}	

	mean_time_to_happen = {
		months = 240
	}
	
	option = {
		name = "frenchitalyconquest.1.a"
		change_culture = toscan
	}
}
