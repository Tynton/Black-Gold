	id = westeurope.2
	title = "westeurope.n.2"
	desc = "westeurope.d.2"
	picture = MODERN_UNREST_eventPicture
	
	#major = yes

	trigger = {
		is_year = 1848
		NOT = {
			is_year = 1850
		}
		NOT = {
			has_reform = parliamentary_reform
		}
		NOT = {
			tag = FRA #will get an alt. event
		}
		NOT = { has_country_modifier = revolution_1848 }
		capital_scope = {
			continent = europe
		}
		NOT = {
			government = theocracy
		}
	}	

	mean_time_to_happen = {
		months = 8
	}
	
	option = {
		name = "westeurope.2.a"
		
		trigger = {
			NOT = {
				OR = {
					culture_group = germanic
					is_subject_of_type = german_confederation_member
				}
			}
		}
		
		add_country_modifier = {
			name = "revolution_1848"
			duration = 1825
		}

			limit = {
				is_capital = yes
			}
			spawn_rebels = {
				type = revolutionary_rebels

			}
		}
		
		ai_chance = { factor = 2 }
	}

		}

#rev1848 france

country_event = {
	id = westeurope.3
	title = "westeurope.n.3"
	desc = "westeurope.d.3"
	picture = MODERN_UNREST_eventPicture
	
	fire_only_once = yes

	trigger = {
		is_year = 1848
		tag = FRA
		NOT = {
			has_reform = parliamentary_reform
		}
	}	

	mean_time_to_happen = {
		months = 4
	}
	
	option = {
		name = "westeurope.3.a" #depose bourbon king
		change_tag = FRR
		change_government = republic
		add_government_reform = parliamentary_reform
		set_government_rank = 3
	}
	
	option = {
		name = "westeurope.3.b" #install con. monarchy
		change_tag = FRE
		add_government_reform = parliamentary_reform
		set_government_rank = 3
	}
	
	option = {
		name = "westeurope.3.c" #do nothing
		add_country_modifier = {
			name = "revolution_1848_france"
			duration = 2920
		}
	}
}
