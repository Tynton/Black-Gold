namespace = dutchgermanyconquest

province_event = {
	id = dutchgermanyconquest.1
	title = "dutchgermanyconquest.n.1"
	desc = "dutchgermanyconquest.d.1"
	picture = ANGRY_MOB_eventPicture

	trigger = {
		AND = {
			culture = swabian
			owner = {
				culture_group = benelux
				#has_country_modifier = funding_cultural_assimilation
			}
		}
		NOT = {
			local_autonomy = 5
			unrest = 1
		}
		owner = {
			OR = {
				stability = 2
				prestige = 50
			}
		}
		has_owner_accepted_culture = no
	}	

	mean_time_to_happen = {
		months = 240
	}
	
	option = {
		name = "dutchgermanyconquest.1.a"
		change_culture = zwabisch
	}
}
