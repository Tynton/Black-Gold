#cores

country_event = {
	id = ukrainecore.1
	title = "ukrainecore.n.1"
	desc = "ukrainecore.d.1"
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = yes
	
	major = yes

	trigger = {
		is_year = 1845
		NOT = {
			culture = ruthenian
		}
		any_owned_province = {
			culture = ruthenian
		}
	}	

	mean_time_to_happen = {
		months = 6
	}
	
	option = {
		name = "ukrainecore.1.a"
		every_province = {
			limit = {
				culture = ruthenian
			}
			add_core = UKR
		}
	}
}
