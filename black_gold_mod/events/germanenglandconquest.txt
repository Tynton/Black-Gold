namespace = germanenglandconquest

province_event = {
	id = germanenglandconquest.1
	title = "germanenglandconquest.n.1"
	desc = "germanenglandconquest.d.1"
	picture = ANGRY_MOB_eventPicture

	trigger = {
		AND = {
			culture = english
			owner = {
				culture_group = germanic
				#has_country_modifier = funding_cultural_assimilation
			}
		}
		OR = {
			owner = {
				ai = no
				government_rank = 3
			}
		}
		NOT = {
			OR = {
				AND = {
					local_autonomy = 5
					unrest = 1
				}
				has_province_modifier = nationalism_unrest_bgm
			}
		}
		owner = {
			OR = {
				stability = 1
				prestige = 50
			}
		}
	}	

	mean_time_to_happen = {
		months = 240
	}
	
	option = {
		name = "germanenglandconquest.1.a"
		change_culture = englisch
	}
}
