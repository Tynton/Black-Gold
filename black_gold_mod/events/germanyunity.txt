country_event = {

	desc = "germanyunity.d.1"

		is_year = 1845

		}

			has_country_flag = gercon_leader
		}

		months = 60
	}

		name = "germanyunity.1.a"
		release = GER

			change_government = republic

			add_government_reform = parliamentary_reform

				who = ROOT
				casus_belli = cb_independence_war
			}
		}
	}

		name = "germanyunity.1.b"
		release = GER

			change_government = republic
			add_government_reform = oligarchy_reform

			create_alliance = ROOT
		}
	}
}

#nations join germany

country_event = {
	id = germanyunity.4
	title = "germanyunity.n.4"
	desc = "germanyunity.d.4"
	picture = ANGRY_MOB_eventPicture

	trigger = {
		exists = GER
		OR = {
			AND = {
				OR = {
					culture_group = germanic
					AND = {
						culture_group = benelux
						is_revolutionary = yes
					}
				}
				NOT = {
					total_development = 500
				}
			}
		}
		NOT = {
			has_country_flag = denied_annexation
		}
		OR = {
			is_subject = no
			is_subject_of = GER
		}
		NOT = {
			truce_with = GER
		}
		capital_scope = {
			continent = europe
		}
	}	

	mean_time_to_happen = {
		months = 3
	}
	
	option = {
		name = "germanyunity.4.a"
		GER = {
			inherit = ROOT
			ROOT = {
				every_owned_province = {
					add_core = GER
				}
			}
		}
		ai_chance = { 
			factor = 1 
			
			modifier = {
				factor = 500
				is_revolutionary = yes
			}
		}
	}
	
	option = {
		name = "germanyunity.4.b"
		GER = {
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_annex
			}
			ROOT = {
				every_owned_province = {
					add_core = GER
				}
			}
		}
		
		set_country_flag = denied_annexation
		
		ai_chance = { 
			factor = 1 
			
			modifier = {
				factor = 10000
				OR = {
					culture = austrian
					culture = swiss
				}
				is_revolutionary = no
			}
		}
	}
}

country_event = {
	id = germanyunity.8
	title = "germanyunity.n.8"
	desc = "germanyunity.d.8"
	picture = HRE_eventPicture
	
	fire_only_once = yes
	
	major = yes

	trigger = {
		has_institution = nationalism_institution
		culture_group = germanic
		is_revolutionary = yes
		NOT = {
			exists = GER
		}
		capital_scope = {
			continent = europe
		}
		is_subject = no
		NOT = {
			is_playing_custom_nation = yes 
		}
		any_subject_country = {
			is_subject_of_type = german_confederation_member
		}
	}	

	mean_time_to_happen = {
		months = 4
	}
	
	option = {
		name = "germanyunity.8.a"
		change_tag = GER
		
		add_stability = -5
		
		every_country = {
			limit = {
				is_subject_of_type = german_confederation_member
			}
			country_event = {
				id = germanyunity.4
				days = 30
			}
		}
		
		north_german_region = {
			add_core = GER
		}
		south_german_region = {
			add_core = GER
		}
	}
