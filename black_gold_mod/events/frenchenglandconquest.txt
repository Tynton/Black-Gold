province_event = {
	id = frenchenglandconquest.1
	title = "frenchenglandconquest.n.1"
	desc = "frenchenglandconquest.d.1"
	picture = ANGRY_MOB_eventPicture
	
	hidden = yes

	trigger = {
		AND = {
			culture = english
			owner = {
				culture_group = french
				#has_country_modifier = funding_cultural_assimilation
			}
		}
		NOT = {
			local_autonomy = 5
			unrest = 1
		}
		OR = {
			owner = {
				ai = no
				government_rank = 3
			}
		}
		NOT = {
			OR = {
				AND = {
					local_autonomy = 5
					unrest = 1
				}
				has_province_modifier = nationalism_unrest_bgm
			}
		}
		owner = {
			OR = {
				stability = 1
				prestige = 50
			}
		}
	}	

	mean_time_to_happen = {
		months = 240
	}
	
	option = {
		name = "frenchenglandconquest.1.a"
		change_culture = anglais
	}
}
