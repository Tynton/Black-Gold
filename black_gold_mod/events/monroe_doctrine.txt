country_event = {
	id = monroe_doctrine.2
	title = "monroe_doctrine.n.2"
	desc = "monroe_doctrine.d.2"
	picture = ANGRY_MOB_eventPicture
	
	hidden = yes

	trigger = {
		ROOT = {
			capital_scope = {
				OR = {
					continent = north_america
					continent = south_america
				}
			}
			is_subject = no
			NOT = {
				has_country_flag = monroe_doctrine_war
			}
		}
		any_country = {
			AND = {
				capital_scope = {
					NOT = {
						OR = {
							continent = north_america
							continent = south_america
						}
						truce_with = USA
					}
				}
				offensive_war_with = ROOT
			}
		}
		NOT = {
			tag = USA
		}
		exists = USA
		
		USA = {
			NOT = {
				war_with = ROOT
			}
		}
		has_global_flag = usa_monroe_doctrine_enacted
	}	

	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "monroe_doctrine.2.a"
		USA = {
			country_event = {
				id = monroe_doctrine.1
			}
		}
		set_country_flag = monroe_doctrine_war
	}
}
