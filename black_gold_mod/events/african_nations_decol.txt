#initial congolese revolt

country_event = {
	id = african_nations_decol.1
	title = "african_nations_decol.n.1"
	desc = "african_nations_decol.d.1"
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = yes
	
	major = yes

	trigger = {
		is_year = 1950
		NOT = {
			culture_group = kongo_group
		}
		NOT = {
			exists = DRC
		}
		development_in_provinces = {
			value = 85
			culture_group = kongo_group
		}
		NOT = {
			tag = DRC
		}
	}	

	mean_time_to_happen = {
		months = 60
	}
	
	option = {
		name = "african_nations_decol.1.a"
		release = DRC
		DRC = {
			change_government = republic
			add_government_reform = oligarchy_reform
			add_government_reform = parliamentary_reform
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_independence_war
			}
		}
	}
	
	option = {
		name = "african_nations_decol.1.b"
		release = DRC
		DRC = {
			change_government = republic
			add_government_reform = oligarchy_reform
			add_government_reform = parliamentary_reform
			create_alliance = ROOT
		}
	}
}

		}

			subject = SAF
		}

		custom_tooltip = dominions_africa_tooltip
	}
}
