namespace = late_colonization

province_event = {
	id = late_colonization.1
	title = "late_colonization.n.1"
	desc = "late_colonization.d.1"
	picture = ANGRY_MOB_eventPicture
	
	hidden = yes

	trigger = {
		all_country = {
			NOT = {
				adm_tech = 34
			}
		}
		continent = africa
		has_port = no
		OR = {
			has_province_modifier = tropical
			has_province_modifier = arid
			has_terrain = marsh
			has_terrain = jungle
			has_terrain = desert
		}
		is_empty = yes
	}	

	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "late_colonization.1.a"
		
		add_province_modifier = {
			name = difficulty_colonizing
			duration = -1
		}
	}
}
