namespace = dutchfranceconquest

province_event = {
	id = dutchfranceconquest.1
	title = "dutchfranceconquest.n.1"
	desc = "dutchfranceconquest.d.1"
	picture = ANGRY_MOB_eventPicture

	trigger = {
		AND = {
			culture = cosmopolitan_french
			owner = {
				culture_group = benelux
				#has_country_modifier = funding_cultural_assimilation
			}
		}
		NOT = {
			local_autonomy = 5
			unrest = 1
		}
		owner = {
			OR = {
				stability = 2
				prestige = 50
			}
		}
		has_owner_accepted_culture = no
	}	

	mean_time_to_happen = {
		months = 240
		
		modifier = {
			factor = 0.5
			
			any_neighbor_province = { 
				culture_group = benelux
			} 
		}
	}
	
	option = {
		name = "dutchfranceconquest.1.a"
		change_culture = frans
	}
}
